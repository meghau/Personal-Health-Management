create or replace procedure proc_signup_insert(dob date,name varchar2,address varchar2,gender varchar2,password varchar2,hsup1 varchar2, auth1 varchar2, hsup2 varchar2, auth2 varchar2, category varchar2)
IS
id number;
BEGIN
    id:=patient_sequence.nextval;
    insert into patient values(id, dob, name, address, gender, password);
    if category='patient' then
        insert into well_patient values(id,hsup1,auth1,hsup2,auth2);
    else
        insert into health_supporter values(id);
    end if;
END;
/
